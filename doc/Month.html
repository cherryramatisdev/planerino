def total_per_owner
  debits_per_owner = []

  debits_not_paid.group_by(&:owner_id).each do |key, value|
    debits_per_owner << {
      owner: Owner.find(key).name,
      total: value.map(&:price).inject(:+)
    }
  end

  debits_per_owner
end
